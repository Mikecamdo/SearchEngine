250template<typename K, typename V>
251V& DSAVLTree<K,V>::insert(const K &x, DSAVLTree::DSAVLNode *&t) {
252 if (t == nullptr) {
253 t = new DSAVLTree::DSAVLNode(x, nullptr, nullptr);
254 return t->data.value;
255 } else if (x < t->data.key) {
256 V& temporary = insert(x, t->left);
257 balance(t);
258 return temporary;
259 } else if (t->data.key < x) {
260 V& temporary = insert(x, t->right);
261 balance(t);
262 return temporary;
263 } else if (t->data.key == x) {
264 return t->data.value;
265 }
266}

286template<typename K, typename V>
287void DSAVLTree<K,V>::balance(DSAVLTree::DSAVLNode *& t) {
288 if (t == nullptr) {
289 return;
290 } if (height(t->left) - height(t->right) > 1) {
291 if (height(t->left->left) >= height(t->left->right)) {
292 rotateWithLeftChild(t);
293 } else {
294 doubleWithLeftChild(t);
295 }
296 } else if (height(t->right) - height(t->left) > 1) {
297 if (height(t->right->right) >= height(t->right->left)) {
298 rotateWithRightChild(t);
299 } else {
300 doubleWithRightChild(t);
301 }
302 }
303 t->height = max(height(t->left), height(t->right)) + 1;
304}

306template<typename K, typename V>
307V &DSAVLTree<K, V>::find(const K& arg, DSAVLTree::DSAVLNode *& t) {
308 if (t == nullptr) {
309 throw std::runtime_error("Value is not in the tree"); //throws an error if the key is not in the tree
310 } else if (arg < t->data.key) {
311 return find(arg, t->left);
312 } else if (t->data.key < arg) {
313 return find(arg, t->right);
314 } else if (arg == t->data.key) {
315 return t->data.value;
316 }

344void DSAVLTree<K,V>::rotateWithLeftChild(DSAVLTree::DSAVLNode*& k2) {
345 DSAVLTree::DSAVLNode* k1 = k2->left;
346 k2->left = k1->right;
347 k1->right = k2;
